BEGIN CalculateAreaBasic
    DECLARE length = 5
    DECLARE width = 3
    DECLARE area
    
    area = length * width
    
    PRINT "Area of rectangle: " + area
END CalculateAreaBasic

BEGIN CalculateAreaWithInput
    DECLARE length, width, area
    
    PRINT "Enter the length of rectangle: "
    INPUT length
    
    PRINT "Enter the width of rectangle: "
    INPUT width
    
    area = length * width
    
    PRINT "Area of rectangle: " + area
END CalculateAreaWithInput
